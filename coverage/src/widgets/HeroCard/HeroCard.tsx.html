import { motion } from 'framer-motion'
import type { Character } from '@entities/character/model'
import { Badge } from '@shared/ui/Badge'
import { Icon } from '@shared/ui/Icon'
import { Tooltip } from '@shared/ui/Tooltip'

interface HeroCardProps {
  character: Character
  index?: number
}

const classVariantMap: Record<string, 'blue' | 'gold' | 'slate' | 'green' | 'red' | 'purple' | 'white'> = {
  Spellcaster: 'blue',
  Defender: 'gold',
  Rogue: 'slate',
  'Nature Guardian': 'green',
  Berserker: 'red',
  'Holy Priest': 'white',
  'Dark Mage': 'purple',
}

const rarityVariantMap: Record<string, 'purple' | 'gold' | 'blue' | 'slate'> = {
  Common: 'slate',
  Rare: 'blue',
  Epic: 'purple',
  Legendary: 'gold',
}

export function HeroCard({ character, index = 0 }: HeroCardProps) {
  const classBadgeVariant = classVariantMap[character.class] || 'blue'

  return (
    <motion.div
      className="group relative overflow-hidden rounded-2xl gold-border"
      initial={{ opacity: 0, y: 40 }}
      whileInView={{ opacity: 1, y: 0 }}
      viewport={{ once: true }}
      transition={{ duration: 0.5, delay: index * 0.1 }}
      whileHover={{ scale: 1.05 }}
    >
      <div className="hover:shadow-[0_0_30px_rgba(236,91,19,0.3)] transition-shadow duration-300">
        {/* Image */}
        <div className="aspect-[3/4] overflow-hidden">
          <motion.img
            alt={character.name}
            className="w-full h-full object-cover"
            src={character.image}
            whileHover={{ scale: 1.1 }}
            transition={{ duration: 0.7 }}
          />
        </div>

        {/* Gradient overlay */}
        <div className="absolute inset-0 bg-gradient-to-t from-background-dark via-transparent to-transparent" />

        {/* Content */}
        <div className="absolute bottom-0 left-0 p-6 space-y-2">
          {/* Badges */}
          <div className="flex gap-2 flex-wrap">
            <Badge variant={classBadgeVariant}>{character.class}</Badge>
            <Badge variant={rarityVariantMap[character.rarity]}>{character.rarity}</Badge>
          </div>

          {/* Name */}
          <h3 className="text-2xl font-black text-white uppercase italic">
            {character.name}
          </h3>

          {/* Skills */}
          <div className="flex gap-4 mt-3 mb-1">
            {character.skills.map((skill) => (
              <Tooltip
                key={skill.id}
                position="top"
                content={
                  <div>
                    <p className="text-white font-bold text-[10px] uppercase mb-1 text-primary">
                      {skill.name}
                    </p>
                    <p className="text-slate-300 text-[9px] leading-tight">
                      {skill.description}
                    </p>
                  </div>
                }
              >
                <div className="flex flex-col items-center group/skill relative cursor-help">
                  <Icon
                    name={skill.icon}
                    className={`${skill.color} text-lg drop-shadow-[0_0_8px_rgba(96,165,250,0.8)]`}
                  />
                  <span className="text-[8px] uppercase font-bold text-blue-400/70 mt-1 opacity-0 group-hover/skill:opacity-100 transition-opacity">
                    {skill.name.split(' ')[0]}
                  </span>
                </div>
              </Tooltip>
            ))}
          </div>

          {/* Description */}
          <p className="text-slate-300 text-xs">{character.description}</p>
        </div>
      </div>
    </motion.div>
  )
}
